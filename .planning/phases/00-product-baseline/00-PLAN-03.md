<action>
Create `ShareGuard.Domain.Tests/Architecture/DependencyTests.cs`:

```csharp
using System.Reflection;
using Xunit;

namespace ShareGuard.Domain.Tests.Architecture;

/// <summary>
/// Validates that the Domain layer maintains its Clean Architecture isolation.
/// Domain must not depend on Application, Infrastructure, or Presentation layers.
/// </summary>
public class DependencyTests
{
    private readonly Assembly _domainAssembly = typeof(ShareGuard.Domain.Marker).Assembly;

    [Fact]
    public void Domain_ShouldNotReference_ApplicationLayer()
    {
        var referencedAssemblies = _domainAssembly.GetReferencedAssemblies();
        Assert.DoesNotContain(referencedAssemblies,
            a => a.Name == "ShareGuard.Application");
    }

    [Fact]
    public void Domain_ShouldNotReference_InfrastructureLayer()
    {
        var referencedAssemblies = _domainAssembly.GetReferencedAssemblies();
        Assert.DoesNotContain(referencedAssemblies,
            a => a.Name == "ShareGuard.Infrastructure");
    }

    [Fact]
    public void Domain_ShouldNotReference_PresentationLayer()
    {
        var referencedAssemblies = _domainAssembly.GetReferencedAssemblies();
        Assert.DoesNotContain(referencedAssemblies,
            a => a.Name == "ShareGuard.App" || a.Name == "Shareguard-wpf");
    }

    [Fact]
    public void Domain_ShouldNotReference_WpfFramework()
    {
        var referencedAssemblies = _domainAssembly.GetReferencedAssemblies();
        Assert.DoesNotContain(referencedAssemblies,
            a => a.Name != null && a.Name.StartsWith("PresentationFramework"));
    }
}
```

This test requires a `Marker` class in the Domain project. Create `ShareGuard.Domain/Marker.cs`:

```csharp
namespace ShareGuard.Domain;

/// <summary>
/// Assembly marker for reflection-based architectural tests.
/// </summary>
public sealed class Marker;
```

The marker class provides a stable type reference for `typeof(Marker).Assembly` in architectural tests without relying on any specific domain entity that might be renamed or moved.
</action>
